export const SearchArticle = ({filter}) =>{
    let searchInput
	const _search=() =>{
		if (searchInput){

			<div className="row" >
            <div className="input-group">
               <input type="text" className="col-md-12" placeholder="Search"

            </div >
        </div>
    )
}
